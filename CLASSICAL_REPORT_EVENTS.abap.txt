REPORT ZCLASSICAL_REPORT_EVENTS.


TYPES : BEGIN OF TY_EMPLOYEE,
          EMP_ID     TYPE ZEMPLOYEE_ID_DATA_ELEMENT_8_10,
          EMP_NAME   TYPE ZEMPLOYEE_NAME_DE_8_10,
          DEPARTMENT TYPE ZDEPARTMENT_DE_8_10,
          MANAGER    TYPE ZMANAGER_DE_8_10,
        END OF TY_EMPLOYEE.


DATA : LT_EMPLOYEE TYPE TABLE OF TY_EMPLOYEE,
       LS_EMPLOYEE TYPE TY_EMPLOYEE.

PARAMETERS : P_EMP_ID TYPE ZEMPLOYEE_ID_DATA_ELEMENT_8_10 MODIF ID PID.
PARAMETERS : P_ERDAT TYPE ERDAT,
             P_ERNAM TYPE ERNAM,
             P_TIME  TYPE TIMS.

DATA LV_EMP_ID TYPE ZEMPLOYEE_ID_DATA_ELEMENT_8_10.
SELECT-OPTIONS S_EMP_ID FOR LV_EMP_ID MODIF ID SID.

PARAMETERS : P_R1 TYPE C RADIOBUTTON GROUP R1,
             P_R2 TYPE C RADIOBUTTON GROUP R1.


INITIALIZATION.

  P_EMP_ID = '101'.
  P_ERDAT = SY-DATUM.
  P_ERNAM = SY-UNAME.
  P_TIME = SY-UZEIT.

  S_EMP_ID-SIGN = 'I'.
  S_EMP_ID-OPTION = 'BT'.
  S_EMP_ID-LOW = '101'.
  S_EMP_ID-HIGH = '105'.
  APPEND S_EMP_ID.

AT SELECTION-SCREEN.

  SELECT  EMP_ID   EMP_NAME DEPARTMENT MANAGER
    FROM ZEMP_TAB_8_10
    INTO TABLE LT_EMPLOYEE
    WHERE EMP_ID IN S_EMP_ID.
  IF LT_EMPLOYEE IS INITIAL.
    MESSAGE E000(ZMESSAGE_EVENT).
  ELSE.
    MESSAGE S001(ZMESSAGE_EVENT).
  ENDIF.

END-OF-SELECTION.
  WRITE : 'PROGRAM END HERE'.


START-OF-SELECTION.
  SELECT  EMP_ID   EMP_NAME DEPARTMENT MANAGER
    FROM ZEMP_TAB_8_10
    INTO TABLE LT_EMPLOYEE
    WHERE EMP_ID IN S_EMP_ID.
  IF LT_EMPLOYEE IS NOT INITIAL.
    WRITE: / LS_EMPLOYEE-EMP_ID, LS_EMPLOYEE-EMP_NAME,LS_EMPLOYEE-DEPARTMENT, LS_EMPLOYEE-MANAGER.
  ENDIF.

TOP-OF-PAGE.
  WRITE 'TOP OF PAGE'.

AT SELECTION-SCREEN OUTPUT.
  IF P_R1 = 'X'.
    LOOP AT SCREEN.
      IF SCREEN-GROUP1 = 'SID'.
        MODIFY SCREEN.
      ENDIF.
    ENDLOOP.
  ELSEIF P_R2 = 'X'.
    LOOP AT SCREEN.
      IF SCREEN-GROUP1 = 'PID'.
        MODIFY SCREEN.
      ENDIF.
    ENDLOOP.
  ENDIF.
